\def\omf@listmode{%
  \def\charged##1##2{(\ensuremath{##1},\ensuremath{##2})}%
  \def\neutral##1{\ensuremath{##1}}%
  \def\tex##1{\ensuremath{\text{\TeX}: ##1}}%
  \def\fortran##1{\texttt{Fortran}: \texttt{##1}}%
  \def\alias##1{alias: \texttt{##1}}%
  \def\anti##1##2{%
    \ifx##1\tex
      \ensuremath{\overline{\text{\TeX}}: ##2}%
    \else
      \ifx##1\fortran
        \ensuremath{\overline{\text{\texttt{Fortran}}}: \texttt{##2}}%
      \else
        \ifx##1\alias
          \ensuremath{\overline{\text{alias}}: ##2}%
        \else
          \errmessage{anti must be followed by tex, fortran or alias!}
        \fi
      \fi
    \fi}}
\newenvironment{omflist}%
  {\omf@listmode}%
  {}%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\def\omf@tab@clearparticle{%
  \global\let\omf@tab@neutral\relax
  \global\let\omf@tab@charged\relax
  \global\let\omf@tab@anti\relax
  \global\let\omf@tab@tex\relax
  \global\let\omf@tab@antitex\relax
  \global\let\omf@tab@fortran\relax
  \global\let\omf@tab@antifortran\relax
  \global\let\omf@tab@alias\relax
  \global\let\omf@tab@antialias\relax}
\def\omf@tab@makerow@{%
  & \ensuremath{\omf@tab@tex}
  & \ensuremath{\omf@tab@antitex}
  & \ensuremath{\omf@tab@fortran}
  & \ensuremath{\omf@tab@antifortran}
  & \ensuremath{\omf@tab@alias}
  & \ensuremath{\omf@tab@antialias} \\
  \omf@tab@clearparticle}
\def\omf@tab@makerow{%
   \ifx\omf@tab@neutral\relax
     \ifx\omf@tab@charged\relax
     \else
       (\ensuremath{\omf@tab@charged},\ensuremath{\omf@tab@anti})%
       \omf@tab@makerow@
     \fi
   \else
     \ensuremath{\omf@tab@neutral}%
     \omf@tab@makerow@
   \fi}
\def\omf@tabularmode{%
  \gdef\charged##1##2{%
    \omf@tab@makerow
    \gdef\omf@tab@charged{##1}%
    \gdef\omf@tab@anti{##2}}
  \gdef\neutral##1{
    \omf@tab@makerow
    \gdef\omf@tab@neutral{##1}}
  \gdef\tex##1{\gdef\omf@tab@tex{##1}}%
  \gdef\fortran##1{\gdef\omf@tab@fortran{##1}}%
  \gdef\alias##1{\gdef\omf@tab@alias{##1}}%
  \gdef\anti##1##2{%
    \ifx##1\tex
      \gdef\omf@tab@antitex{##1}%
    \else
      \ifx##1\fortran
        \gdef\omf@tab@antifortran{##1}%
      \else
        \ifx##1\alias
          \gdef\omf@tab@antialias{##1}%
        \else
          \errmessage{anti must be followed by tex, fortran or alias!}
        \fi
      \fi
    \fi}
  \gdef\anti##1##2{\relax}%
  \omf@tab@clearparticle}
\newenvironment{omftable}%
  {\omf@tabularmode
   \begin{tabular}{c|cc|cc|cc}%
   Name
    & \multicolumn{2}{|l|}{\TeX}
    & \multicolumn{2}{|l|}{\texttt{Fortran}}
    & \multicolumn{2}{|l}{alias(es)}\\\hline}
  {\omf@tab@makerow
   \end{tabular}}%
