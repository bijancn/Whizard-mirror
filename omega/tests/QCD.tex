\newcommand{\charged}[2]{(\ensuremath{#1},\ensuremath{#2})}
\newcommand{\neutral}[1]{\ensuremath{#1}}
\newcommand{\tex}[1]{\ensuremath{\text{\TeX}: #1}}
\newcommand{\fortran}[1]{\texttt{Fortran}: \texttt{#1}}
\newcommand{\alias}[1]{alias: \texttt{#1}}
\def\anti#1#2{%
  \ifx#1\tex
    \ensuremath{\overline{\text{\TeX}}: #2}%
  \else
    \ifx#1\fortran
      \ensuremath{\overline{\text{\texttt{Fortran}}}: \texttt{#2}}%
    \else
      \ifx#1\alias
        \ensuremath{\overline{\text{alias}}: #2}%
      \else
        \errmessage{anti must be followed by tex, fortran or alias!}
      \fi
    \fi
  \fi}
